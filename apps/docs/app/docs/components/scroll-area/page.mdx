## Installation

```bash
npx r-ui add scroll-area
```

## Usage

```tsx
import {
  ScrollArea,
  ScrollAreaViewport,
  ScrollAreaScrollbar,
  ScrollAreaThumb,
} from '@r-ui/react-native'

export default function MyComponent() {
  return (
    <ScrollArea style={{ height: 200, width: 300 }}>
      <ScrollAreaViewport>
        {items.map((item) => (
          <View key={item.id}>
            <Text>{item.name}</Text>
          </View>
        ))}
      </ScrollAreaViewport>
      <ScrollAreaScrollbar orientation="vertical">
        <ScrollAreaThumb />
      </ScrollAreaScrollbar>
    </ScrollArea>
  )
}
```

## Horizontal Usage

```tsx
<ScrollArea horizontal style={{ height: 200, width: 300 }}>
  <ScrollAreaViewport contentContainerStyle={{ flexDirection: 'row' }}>
    {images.map((img) => (
      <Image key={img.id} source={img.source} style={{ width: 150, height: 150 }} />
    ))}
  </ScrollAreaViewport>
  <ScrollAreaScrollbar orientation="horizontal">
    <ScrollAreaThumb orientation="horizontal" />
  </ScrollAreaScrollbar>
</ScrollArea>
```

## With Force Visible Scrollbar

```tsx
<ScrollArea style={{ height: 200 }}>
  <ScrollAreaViewport>
    {/* Content */}
  </ScrollAreaViewport>
  <ScrollAreaScrollbar forceVisible>
    <ScrollAreaThumb />
  </ScrollAreaScrollbar>
</ScrollArea>
```

## Simple Usage (Native Indicators)

For simple cases, you can use ScrollArea with just the viewport and native scroll indicators:

```tsx
<ScrollArea showsIndicator style={{ height: 300 }}>
  <ScrollAreaViewport>
    {items.map((item) => (
      <Text key={item.id}>{item.name}</Text>
    ))}
  </ScrollAreaViewport>
</ScrollArea>
```

## Both Scrollbars

When content scrolls both horizontally and vertically, use the corner component:

```tsx
<ScrollArea style={{ height: 300, width: 300 }}>
  <ScrollAreaViewport>
    <View style={{ width: 600, height: 600 }}>
      {/* Large content */}
    </View>
  </ScrollAreaViewport>
  <ScrollAreaScrollbar orientation="vertical">
    <ScrollAreaThumb />
  </ScrollAreaScrollbar>
  <ScrollAreaScrollbar orientation="horizontal">
    <ScrollAreaThumb orientation="horizontal" />
  </ScrollAreaScrollbar>
  <ScrollAreaCorner />
</ScrollArea>
```

## Props

### ScrollArea

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | - | Content to render |
| `horizontal` | `boolean` | `false` | Enable horizontal scrolling |
| `showsIndicator` | `boolean` | `true` | Show native scroll indicators |
| `bounces` | `boolean` | `true` | Enable bounce effect |
| `style` | `ViewStyle` | - | Additional styles |

### ScrollAreaViewport

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | - | Content to render |
| `contentContainerStyle` | `ViewStyle` | - | Styles for the content container |
| `style` | `ViewStyle` | - | Additional styles |

### ScrollAreaScrollbar

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `orientation` | `'vertical' \| 'horizontal'` | `'vertical'` | Scrollbar orientation |
| `forceVisible` | `boolean` | `false` | Keep scrollbar always visible |
| `children` | `ReactNode` | - | Scrollbar thumb |
| `style` | `ViewStyle` | - | Additional styles |

### ScrollAreaThumb

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `orientation` | `'vertical' \| 'horizontal'` | `'vertical'` | Thumb orientation |
| `style` | `ViewStyle` | - | Additional styles |

### ScrollAreaCorner

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `style` | `ViewStyle` | - | Additional styles |

## Features

- Custom styled scrollbar matching r/ui theme
- Auto-hide scrollbar when not scrolling (fades out after 1 second)
- Horizontal and vertical scrolling support
- Smooth thumb animation based on scroll position
- Thumb size proportional to content vs viewport ratio
- Native scroll indicators available as fallback
- Works with both touch (mobile) and mouse (web)

## Accessibility

- Uses native ScrollView for proper accessibility support
- Maintains scroll position and momentum
- Supports keyboard navigation on web
