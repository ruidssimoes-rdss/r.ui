## Installation

```bash
npx r-ui add date-range-picker
```

## Usage

```tsx
import {
  DateRangePicker,
  DateRangePickerTrigger,
  DateRangePickerContent,
  DateRangePickerHeader,
  DateRangePickerCalendar,
  DateRangePickerPresets,
  DateRange,
} from '@r-ui/react-native'

export default function MyComponent() {
  const [range, setRange] = useState<DateRange>({ start: null, end: null })

  return (
    <DateRangePicker value={range} onValueChange={setRange}>
      <DateRangePickerTrigger placeholder="Select date range..." />
      <DateRangePickerContent>
        <View style={{ flexDirection: 'row' }}>
          <View>
            <DateRangePickerHeader />
            <DateRangePickerCalendar />
          </View>
          <DateRangePickerPresets />
        </View>
      </DateRangePickerContent>
    </DateRangePicker>
  )
}
```

## Features

- **Range selection**: Click to select start, click again for end
- **Visual highlighting**: Range between dates is highlighted
- **Preset ranges**: Quick-select Today, Last 7 days, This month, etc.
- **Date constraints**: Set min/max dates or disable specific dates
- **Month/year navigation**: Click header to switch between views
- **Accessible**: Full ARIA support

## Props

### DateRangePicker

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `DateRange` | - | Selected date range (controlled) |
| `onValueChange` | `(range: DateRange) => void` | - | Called when range changes |
| `defaultValue` | `DateRange` | `{ start: null, end: null }` | Default range (uncontrolled) |
| `disabled` | `boolean` | `false` | Disable the picker |
| `placeholder` | `string` | `'Select dates...'` | Placeholder text |
| `format` | `string` | `'MMM d, yyyy'` | Date display format |
| `minDate` | `Date` | - | Minimum selectable date |
| `maxDate` | `Date` | - | Maximum selectable date |
| `disabledDates` | `(date: Date) => boolean` | - | Function to disable dates |
| `presets` | `PresetRange[] \| false` | Default presets | Custom presets or false to hide |
| `open` | `boolean` | - | Controlled open state |
| `onOpenChange` | `(open: boolean) => void` | - | Called when open state changes |

### DateRangePickerTrigger

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `placeholder` | `string` | - | Override placeholder text |
| `style` | `ViewStyle` | - | Additional styles |

### DateRange Type

```tsx
interface DateRange {
  start: Date | null
  end: Date | null
}
```

### PresetRange Type

```tsx
interface PresetRange {
  label: string
  value: DateRange
}
```

### Default Presets

The following presets are included by default:

- **Today** - Current date
- **Yesterday** - Previous day
- **Last 7 days** - Past week including today
- **Last 30 days** - Past month including today
- **This month** - First to last day of current month
- **Last month** - First to last day of previous month

### Custom Presets

```tsx

  },
```
